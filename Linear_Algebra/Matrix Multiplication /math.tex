\begin{abstract}
    This papers presents a rigorous study of composition of linear transformations and multiplication of their corresponding coordinate matrices. Here we establish that a matrix multiplication is the exact algebraic representation of the composition of their corresponding transformations.
\end{abstract}

\section{Introduction}
We know a transformation $T: V \to W$ describes how one space maps into another. The natural progression of this concept arises when we consider a second linear transformation, $U: W \to Z$. By applying U to the image of T, we establish a chained operation known as composition.

\section{Composition of Linear Transformations}
Composition of linear transformations is just like composite functions. Just as we show $f \circ g (x) = f(g(x))$, similarly in linear algebra we represent our composition of transformations as $UT(x)$ not $U \circ T$ and we define $UT(x) = U(T(x))$.

\begin{theorem}
    Let $V, W$ and $Z$ be vector spaces over the same field $F$, and let $T:V \to W$ and $U:W \to Z$ be linear. Then $UT:V \to Z$ is linear.
\end{theorem}

\begin{proof}
    Since $T$ and $U$ are linear, and let $x,y \in V$ and $a \in F$. Then
    \begin{align*}
        UT(ax+y) &= U(T(ax+y)) = U(aT(x)+T(y)) \\
        & \quad = aU(T(x)) +U(T(y)) = a (UT)(x) + (UT)(y)
    \end{align*}
    Hence, $UT$ is linear 
\end{proof}

\begin{theorem}
    Let $V$ be a vector space. Let $T,U_1, U_2 \in \mathscr{L}(V)$. Then
    \begin{enumerate}
        \item $T(U_1+U_2)=TU_1+TU_2$ and $(U_1+U_2)T=U_1T+U_2T$.
        \item $T(U_1U_2)=(TU_1)U_2$.
        \item $TI=IT=T$.
        \item $a(U_1U_2)=(aU_1)U_2=U_1(aU_2)$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    Given, $T,U_1, U_2 \in \mathscr{L}(V)$. Then $T,U_1 \ \& \ U_2$ are linear. Then for some $v \in V$, we have
    \begin{enumerate}
        \item \begin{align*}
            (T(U_1+U_2))(v) &= T((U_1+U_2)(v)) = T(U_1(v)+U_2(v)) \\
            &= T(U_1(v))+T(U_2(v)) = TU_1(v)+TU_2(v)
        \end{align*}
        Also
        \begin{align*}
            ((U_1+U_2)T)(v) &= (U_1+U_2)(T(v)) = U_1(T(v))+U_2(T(v)) \\
            &= U_1T(v)+U_2T(v)
        \end{align*}

        \item \begin{align*}
            (T(U_1U_2))(v) = T((U_1U_2)(v)) = T(U_1(U_2(v))) = (TU_1)(U_2(v))
        \end{align*}

        \item \[
        (TI)(v) = T(I(v)) = T(v)
        \]
        Also, \[
        (IT)(v) = I(T(v)) = T(v)
        \]

        \item \begin{align*}
            (a(U_1U_2))(v) = a((U_1U_2)(v)) = a(U_1(U_2(v)))
        \end{align*}
        Now \[
        ((aU_1)U_2)(v) = (aU_1)(U_2(v)) = a(U_1(U_2(v)))
        \]
        Thus, $a(U_1U_2) = (aU_1)U_2$.

        \noindent Also \[
        (U_1(aU_2))(v) = U_1((aU_2)(v))=U_1(a(U_2(v))) = a(U_1(U_2(v)))
        \]
        Thus, $a(U_1U_2)=U_1(aU_2)$
    \end{enumerate}
\end{proof}

\section{Matrix Multiplication}
Let $T:V \to W$ and $U:W \to Z$ be linear transformations, and let $A = [U]_\beta^\gamma$ and $B = [T]_\alpha^\beta$ where $\alpha = \{v_1,v_2,\ldots,v_n\}$ and $\beta = \{w_1,w_2,\ldots,w_m\}$ and $\gamma = \{z_1,z_2,\ldots,z_p\}$ are ordered bases for $V,W$ and $Z$, respectively. Then the product of two matrices are defined as $AB =[UT]_\alpha^\gamma$.

For $1 \le j \le n$, we have
\begin{align*}
    UT(v_j) &= U(T(v_j)) = U\left(\sum_{k=1}^m B_{kj}w_k\right) = \sum_{k=1}^mB_{kj}w_k \\
    &= \sum_{k=1}^m B_{kj}\left(\sum_{i=1}^p A_{ik}z_i\right) = \sum_{i=1}^p\left(\sum_{k=1}^m A_{ik}B_{kj}\right)z_i \\
    &= \sum_{i=1}^p C_{ij}z_i
\end{align*}
where
\[
C_{ij} = \sum_{k=1}^m A_{ik}B_{kj}
\]

\begin{definition}
    Let $A$ be an $m \times n$ matrix and $B$ be an $n \times p$ matrix. We define the product of $A$ and $B$, denoted $AB$, to be the $m \times p$ matrix such that
    \[
    (AB)_{ij} = \sum_{k=1}^n A_{ik}B_{kj} \quad \text{for} \ 1 \le i \le m, \ 1 \le j \le p
    \]
\end{definition}

\begin{theorem}
    Let $V, W$ and $Z$ be finite dimensional vector spaces with ordered bases $\alpha, \beta$ and $\gamma$, respectively. Let $T: V \to W$ and $U: W \to Z$ be linear. Then,
\[ [UT]_{\alpha}^{\gamma} = [U]_{\beta}^{\gamma} [T]_{\alpha}^{\beta} \]
\end{theorem}

\begin{proof}
Using the above notation, we have,
\[ [UT]_{\alpha}^{\gamma} = AB \]
where $A = [U]_{\beta}^{\gamma}$ and $B = [T]_{\alpha}^{\beta}$.

\noindent Thus,
\[ [UT]_{\alpha}^{\gamma} = AB = [U]_{\beta}^{\gamma} [T]_{\alpha}^{\beta} \]
\end{proof}

\begin{corollary}
    Let $V$ be a finite dimensional vector space with an ordered basis $\beta$. Let $T,U \in \mathscr{L}(V)$. Then
    \[[UT]_\beta = [U]_\beta [T]_\beta\]
\end{corollary}

\begin{proof}
    This follows directly from previous theorem if $V=W=Z$ and $\alpha = \beta = \gamma$. Then,
    \[
    [UT]_\beta = [U]_\beta [T]_\beta
    \]
\end{proof}

\begin{definition}
    We define the Kronecker delta $\delta_{ij}$ by $\delta_{ij} = 1$ if $i=j$ and $\delta_{ij} = 0$ if $i \ne j$. The $n \times n$ identity matrix $I_n$ is defined by $(I_n)_{ij} = \delta_{ij}$
\end{definition}

\begin{theorem}
    Let $A$ be an $m \times n$ matrix, $B$ and $C$ be $n \times p$ matrices, and $D$ and $E$ be $q \times m$ matrices. Then
\begin{enumerate}
    \item $A(B+C) = AB + AC$ and $(D+E)A = DA + EA$
    \item $a(AB) = (aA)B = A(aB)$ for any scalar $a$.
    \item $I_m A = A = A I_n$
    \item If $V$ is an $n$-dimensional vector space with an ordered basis $\beta$, then $[I_V]_\beta = I_n$.
\end{enumerate}
\end{theorem}

\begin{proof}
    \begin{enumerate}
        \item \begin{align*}
           [A(B+C)]_{ij} &= \sum_{k=1}^{n} A_{ik}(B+C)_{kj} = \sum_{k=1}^{n} A_{ik}(B_{kj} + C_{kj}) \\
&= \sum_{k=1}^{n} (A_{ik}B_{kj} + A_{ik}C_{kj}) = \sum_{k=1}^{n} A_{ik}B_{kj} + \sum_{k=1}^{n} A_{ik}C_{kj} \\
&= (AB)_{ij} + (AC)_{ij} = [AB + AC]_{ij}
\end{align*}
\[ \therefore A(B+C) = AB + AC \]

\noindent Again,
\begin{align*}
[(D+E)A]_{ij} &= \sum_{k=1}^{m} (D+E)_{ik} A_{kj} = \sum_{k=1}^{m} (D_{ik} + E_{ik}) A_{kj} \\
&= \sum_{k=1}^{m} (D_{ik}A_{kj} + E_{ik}A_{kj}) = (DA)_{ij} + (EA)_{ij} \\
&= [DA + EA]_{ij}
\end{align*}
\[ \therefore (D+E)A = DA + EA \]

\item We have,
\begin{align*}
[a(AB)]_{ij} &= a[AB]_{ij} = a \sum_{k=1}^{n} A_{ik}B_{kj} = \sum_{k=1}^{n} a(A_{ik}B_{kj}) \\
&= \sum_{k=1}^{n} (aA)_{ik} B_{kj} = [(aA)B]_{ij} \\
\text{or } \qquad &= \sum_{k=1}^{n} A_{ik} (aB)_{kj} = [A(aB)]_{ij}
\end{align*}
So,
\[ a(AB) = (aA)B = A(aB) \]

\item We have,
\[ [I_m A]_{ij} = \sum_{k=1}^{m} (I_m)_{ik} A_{kj} = \sum_{k=1}^{m} \delta_{ik} A_{kj} = A_{ij} \]
Again,
\[ [AI_n]_{ij} = \sum_{k=1}^{n} A_{ik} (I_n)_{kj} = \sum_{k=1}^{n} A_{ik} \delta_{kj} = A_{ij} \]
So
\[ I_m A = A = AI_n \]

\item $I_V(v_j) = v_j$. Thus,
\[ I_V(v_j) = 0v_1 + 0v_2 + \dots + 1v_j + \dots + 0v_n \]
\noindent or we can write,
\[ I_V(v_j) = \sum_{i=1}^{n} \delta_{ij} v_i \]
where $\delta_{ij} = 1$ if $i=j$ and $\delta_{ij} = 0$ if $i \neq j$, which is the definition of Kronecker delta. Thus, $[I_V]_\beta = I_n$.
    \end{enumerate}
\end{proof}

\begin{corollary}
    Let $A$ be an $ m \times n$ matrix and $B_1,B_2,\ldots,B_n$ be $n \times p$ matrices and $C_1,C_2,\ldots,C_p$ be $q \times m$ matrices, and $a_1,a_2,\ldots,a_k$ be scalars. Then
    \[
    A\left(\sum_{i=1}^k a_iB_i\right) = \sum_{i=1}^k a_i(AB_i) \quad \text{and} \quad \left(\sum_{i=1}^k a_iC_i\right)A = \sum_{i=1}^k a_i(C_iA)
    \]
\end{corollary}

\begin{proof}
    Proving $A\left(\displaystyle\sum_{i=1}^k a_iB_i\right) = \displaystyle\sum_{i=1}^k a_i(AB_i)$:

    \noindent Let $k = 1$ then, $A(a_1B_1)=a_1(AB_1)$ by Theorem-3.2(2).

    \noindent Let $k=2$ then, $A(a_1B_1+a_2B_2)=a_1(AB_1)+a_2(AB_2)$ by Theorem-3.2(1),(2). 

    \noindent Thus the theorem is true for $k=1,2$. Now, let the equation be true for an arbitrary $k=m$.Then,
    \begin{align*}
        A\left(\sum_{i=1}^m a_iB_i\right) &= \sum_{i=1}^m a_i(AB_i)\\
        \implies A\left(\sum_{i=1}^m a_iB_i + a_{m+1}B_{m+1}-a_{m+1}B_{m+1}\right) &= \sum_{i=1}^m a_i(AB_i) \\
        \implies A\left(\sum_{i=1}^{m+1} a_iB_i\right) - a_{m+1}(AB_{m+1}) &= \sum_{i=1}^m a_i(AB_i)\\
        \implies A\left(\sum_{i=1}^{m+1} a_iB_i\right) &= \sum_{i=1}^m a_i(AB_i) + a_{m+1}(AB_{m+1})\\
        &= \sum_{i=1}^{m+1} a_i(AB_i)
    \end{align*}
    Thus the theorem is true for any $k$ by the method of induction.

    \vspace{0.5cm}

    \noindent Proving $\left(\displaystyle\sum_{i=1}^k a_iC_i\right)A = \displaystyle\sum_{i=1}^k a_i(C_iA)$:

    \noindent Let $k = 1$ then, $(a_1C_1)A=a_1(C_1A)$ by Theorem-3.2(2).

    \noindent Let $k=2$ then, $(a_1C_1+a_2C_2)A=a_1(C_1A)+a_2(C_2A)$ by Theorem-3.2(1),(2). 

    \noindent Thus the theorem is true for $k=1,2$. Now, let the equation be true for an arbitrary $k=m$.Then,
    \begin{align*}
        \left(\sum_{i=1}^m a_iC_i\right)A &= \sum_{i=1}^m a_i(C_iA)\\
        \implies \left(\sum_{i=1}^m a_iC_i + a_{m+1}C_{m+1}-a_{m+1}C_{m+1}\right)A &= \sum_{i=1}^m a_i(C_iA) \\
        \implies \left(\sum_{i=1}^{m+1} a_iC_i\right)A - a_{m+1}(C_{m+1}A) &= \sum_{i=1}^m a_i(C_iA)\\
        \implies \left(\sum_{i=1}^{m+1} a_iC_i\right)A &= \sum_{i=1}^m a_i(C_iA) + a_{m+1}(C_{m+1}A)\\
        &= \sum_{i=1}^{m+1} a_i(C_iA)
    \end{align*}
    Thus the theorem is true for any $k$ by the method of induction.
\end{proof}

\begin{theorem}
    Let $A$ be an $m \times n$ matrix and $B$ be an $n \times p$ matrix. For each $j(1 \le j \le p)$ let $u_j$ and $v_j$ denote the j-th columns of $AB$ and $B$, respectively. Then
    \begin{enumerate}
        \item $u_j = Av_j$
        \item $v_j = Be_j$, where $e_j$ is the j-th standard vector of $F^p$
    \end{enumerate}
\end{theorem}

\begin{proof}
    Given,
    \[
    u_j = \begin{bmatrix}
        (AB)_{1j} \\ (AB)_{2j} \\ \vdots \\ (AB)_{mj}
    \end{bmatrix} \quad \text{and} \quad 
    v_j = \begin{bmatrix}
        B_{1j} \\ B_{2j} \\ \vdots \\ B_{nj}
    \end{bmatrix}
    \]
    \begin{enumerate}
        \item We have
        \begin{align*}
        u_j &= \begin{bmatrix}
            (AB)_{1j} \\ (AB)_{2j} \\ \vdots \\ (AB)_{mj}
        \end{bmatrix} = \begin{bmatrix}
            \displaystyle\sum_{k=1}^n A_{1k}B_{kj} \\
            \displaystyle\sum_{k=1}^n A_{2k}B_{kj} \\
            \vdots \\
            \displaystyle\sum_{k=1}^n A_{nk}B_{kj} \\
        \end{bmatrix} = \begin{bmatrix}
            A_{11}B_{1j}+A_{12}B_{2j}+\ldots+A_{1n}B_{nj} \\
            A_{21}B_{1j}+A_{22}B_{2j}+\ldots+A_{2n}B_{nj} \\
            \vdots \\
            A_{m1}B_{1j}+A_{m2}B_{2j}+\ldots+A_{mn}B_{nj} \\
        \end{bmatrix} \\
        &= \begin{bmatrix}
            A_{11} & A_{12} & \ldots & A_{1n} \\
            A_{21} & A_{22} & \ldots & A_{3n} \\
            \vdots & \vdots & \ddots & \vdots \\
            A_{m1} & A_{m2} & \ldots & A_{mn} \\
        \end{bmatrix} . \begin{bmatrix}
            B_{1j} \\ B_{2j} \\ \vdots \\ B_{nj}
        \end{bmatrix} = Av_j
        \end{align*}

        \item Since,$e_j$ is the j-th standard vector of $F^p$, we have
        \[e_j = (0,0,\ldots,0,1,0,\ldots,0)\]
        Again
        \[B_{ij} = (BI_n)_{ij} = \sum_{k=1}^pB_{ik}\delta_{kj}\]
        Here, \[(e_j)_i = \delta_{ij} = \begin{cases}
            0 & i \ne j \\
            1 & i=j
        \end{cases}\]
        \[\therefore \ B_{ij} = \sum_{k=1}^pB_{ik}(e_j)_k = (Be_j)_i\]
        \begin{align*}
            Be_j &= \begin{bmatrix}
                B_{11}(0)+B_{12}(0)+\ldots+B_{1j}(1)+\ldots+B_{1p}(0) \\
                B_{21}(0)+B_{22}(0)+\ldots+B_{2j}(1)+\ldots+B_{2p}(0) \\
                \vdots \\
                B_{n1}(0)+B_{n2}(0)+\ldots+B_{nj}(1)+\ldots+B_{np}(0)
            \end{bmatrix} = \begin{bmatrix}
            B_{1j} \\ B_{2j} \\ \vdots \\ B_{nj}
        \end{bmatrix} = v_j
        \end{align*}
    \end{enumerate}
\end{proof}

\begin{theorem}
    Assuming the notation of the previous theorem.
    \begin{enumerate}
        \item Suppose the $z$ is a column vector in $F^p$ then $Bz$ is a linear combination of columns of $B$,\textit{i.e.}, if $z = (a_1,a_2,\ldots,a_p)^t$ then 
        \[Bz = \sum_{j=1}^p a_jv_j\]
        \item Extend (1) to column $j$ of $AB$ is a linear combination of the columns of $A$ with the coefficients in the linear combination being the entries of column $j$ of $B$.
        \item For any row vector $w \in F^m$, prove that $wA$ is a linear combination of the rows of $A$ with the coefficients in the linear combination being the coordinates of w.
        \item Extend (3) to row $i$ of $AB$ is a linear combination of the rows of $B$ with the coefficients in the linear combination being the entries of row $i$ of $A$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    From previous theorem, we have, A is an $m \times n$ matrix and $B$ is an $n \times p$ matrix. If $1 \le j \le p$ then
    \[
    u_j = \begin{bmatrix}
        (AB)_{1j} \\ (AB)_{2j} \\ \vdots \\ (AB)_{mj}
    \end{bmatrix} \quad \text{and} \quad 
    v_j = \begin{bmatrix}
        B_{1j} \\ B_{2j} \\ \vdots \\ B_{nj}
    \end{bmatrix}
    \]
    Also, $u_j=Av_j$ and $v_j=Be_j$, where $e_j$ is standard j-th vector of $F^p$.

    \begin{enumerate}
        \item \[
        B=\begin{bmatrix}
            B_{11} & B_{12} & \ldots & B_{1p} \\
            B_{21} & B_{22} & \ldots & B_{2p} \\
            \vdots & \vdots & \ddots & \ldots \\
            B_{n1} & B_{n2} & \ldots & B_{np} \\
        \end{bmatrix} \quad \text{and} \quad z = \begin{bmatrix}
            a_1 \\ a_2 \\ \vdots \\ a_p
        \end{bmatrix}
        \]
        Then
        \[
        Bz = \begin{bmatrix}
            a_1B_{11}+a_2B_{12}+\ldots+a_pB_{1p}\\
            a_1B_{21}+a_2B_{22}+\ldots+a_pB_{2p}\\
            \vdots \\
            a_1B_{n1}+a_2B_{n2}+\ldots+a_pB_{np}\\
        \end{bmatrix}
        \]
        By matrix addition and scalar multiplication of matrices, we have
        \begin{align*}
            Bz &= \begin{bmatrix}
                a_1B_{11} \\ a_1B_{21} \\ \vdots \\ a_1B_{n1}
            \end{bmatrix} + \begin{bmatrix}
                a_2B_{12} \\ a_2B_{22} \\ \vdots \\ a_2B_{n2}
            \end{bmatrix} + \ldots + \begin{bmatrix}
                a_pB_{1p} \\ a_pB_{2p} \\ \vdots \\ a_pB_{np}
            \end{bmatrix} \\
            &= a_1\begin{bmatrix}
                B_{11} \\ B_{21} \\ \vdots \\ B_{n1}
            \end{bmatrix} + a_2\begin{bmatrix}
                B_{12} \\ B_{22} \\ \vdots \\ B_{n2}
            \end{bmatrix} + \ldots + a_p\begin{bmatrix}
                B_{1p} \\ B_{2p} \\ \vdots \\ B_{np}
            \end{bmatrix} \\
            &= a_1v_1+a_2v_2+\ldots+a_pv_p \\
            &= \sum_{j=1}^pa_jv_j
        \end{align*}
        Thus we prove that if we multiply a vector to a matrix then the entries of the vector acts as coefficients to the linear combination of the columns of the matrix.

        \item Let $w_1,a_2,\ldots,a_n$ be the columns of matrix $A$, \textit{i.e.},
        \[
        a_j = \begin{bmatrix}
            A_{1j} \\ A_{2j} \\ \vdots \\ A_{mj}
        \end{bmatrix} \quad 1 \le j \le n.
        \]
        Also, we have $v_j$ to be the j-th column of $B$ and also that $u_j = Av_j$. Now from part (1), we have if we multiply a matrix with a vector then the entries of vector act as coefficients in the linear combination of the columns of matrix. Thus,
        \[
        u_j = B_{1j}a_1 + B_{2j}a_2+ \ldots + B_{nj}a_n = \sum_{k=1}^nB_{kj}a_k
        \]
        Thus we prove that j-th column of $AB$ is a linear combination of columns of $A$ with j-th column of $B$ as the coefficients.

        \item We have $w$ is a row vector in $F^m$ then we can write $w$ as \[
        w = \begin{bmatrix}
            w_1 & w_2 & \ldots & w_m
        \end{bmatrix}
        \]
        Also we have $A$ is an $m \times n$ matrix. Then $wA$ is defined and,
        \begin{align*}
            wA &= \begin{bmatrix}
                w_1 & w_2 & \ldots & w_m
            \end{bmatrix} \cdot \begin{bmatrix}
                A_{11} & A_{12} & \ldots & A_{1n} \\
                A_{21} & A_{22} & \ldots & A_{3n} \\
                \vdots & \vdots & \ddots & \vdots \\
                A_{m1} & A_{m2} & \ldots & A_{mn} \\
            \end{bmatrix} \\
            &= \begin{bmatrix}
                w_1A_{11}+w_2A_{21}+\ldots+w_mA_{m1} & w_1A_{11}+w_2A_{21}+\ldots+w_mA_{m1} \\ & \quad \quad  \ldots w_1A_{11}+w_2A_{21}+\ldots+w_mA_{m1}
            \end{bmatrix}\\
            &= \begin{bmatrix}
                w_1A_{11} & w_1A_{12} & \ldots & w_1A_{1n}
            \end{bmatrix} + \begin{bmatrix}
                w_2A_{21} & w_2A_{22} & \ldots & w_2A_{2n}
            \end{bmatrix} \\
            & \quad \quad \ldots + \begin{bmatrix}
                w_mA_{m1} & w_mA_{m2} & \ldots & w_mA_{mn}
            \end{bmatrix} \\
            &= w_1\begin{bmatrix}
                A_{11} & A_{12} & \ldots & A_{1n}
            \end{bmatrix} + w_2\begin{bmatrix}
                A_{21} & A_{22} & \ldots & A_{2n}
            \end{bmatrix} \\
            & \quad \quad \ldots + w_m\begin{bmatrix}
                A_{m1} & A_{m2} & \ldots & A_{mn}
            \end{bmatrix} \\
        \end{align*}
        If we take $A_1,A_2, \dots, A_n$ to be the rows of $A$, then
        \[
        wA = w_1A_1+w_2A_2+\ldots+w_mA_m = \sum_{k=1}^mw_kA_k
        \]
        Thus we prove that $wA$ is a linear combination of the rows of $A$ with coefficients being the coordinates of $w$.

        \item let $A_i$ and $B_j$ be the rows of $A$ and $B$ respectively, such that
        \[
        A_i = \begin{bmatrix}
            A_{i1} & A_{i2} & \ldots & A_{in}
        \end{bmatrix} \quad \text{and} \quad B_j = \begin{bmatrix}
            B_{j1} & B_{j2} & \ldots & B_{jp}
        \end{bmatrix}
        \]
        for $1 \le i \le m$ and $1 \le j \le n$, respectively. Here
        \begin{align*}
            AB &= \begin{bmatrix}
                A_{11} & A_{12} & \ldots & A_{1n}
            \end{bmatrix}B + \begin{bmatrix}
                A_{21} & A_{22} & \ldots & A_{2n}
            \end{bmatrix}B \\
            & \quad \quad \ldots + \begin{bmatrix}
                A_{m1} & A_{m2} & \ldots & A_{mn}
            \end{bmatrix}B
        \end{align*}
        Since, in part (3), we proved that if we multiply a row vector with a matrix then it forms a linear combination of rows of matrix then it forms a linear combination of rows of matrix with the coefficients being the coordinates of row vector. Then,
        \[
        AB = \begin{bmatrix}
            \displaystyle\sum_{k=1}^n A_{1k}B_k \\
            \displaystyle\sum_{k=1}^n A_{2k}B_k \\
            \vdots \\
            \displaystyle\sum_{k=1}^n A_{mk}B_k
        \end{bmatrix} \begin{matrix}
            \to \text{1st row} \\ \\
            \to \text{2nd row} \\
            \\ \\ \\
            \to \text{m-th row}
        \end{matrix}
        \]
        Thus
        \[\text{i-th row of } AB = \sum_{k=1}^n A_{ik}B_k\]
        Hence we proved that i-th row of $AB$ is a linear combination of rows of $B$ with coefficients being the entries of i-th rows of $A$.
    \end{enumerate}
\end{proof}

\begin{theorem}
    Let $V$ and $W$ be finite dimensional vector spaces having ordered bases $\beta$ and $\gamma$, respectively, and let $T:V \to W$ be linear. Then for each $u \in V$, we have
    \[
    [T(u)]_\gamma = [T]_\beta^\gamma [u]_\beta
    \]
\end{theorem}

\begin{proof}
    Let $\beta = \{v_1, v_2, \dots, v_n\}$ and $\gamma = \{w_1, w_2, \dots, w_m\}$ be the bases of $V$ and $W$ respectively. Since $u \in V$ then,
    \[ u = \sum_{j=1}^{n} c_j v_j \text{ for } c_j \in F \]
    Then,
    \[ [u]_\beta = \begin{bmatrix} c_1 \\ c_2 \\ \vdots \\ c_n \end{bmatrix} \]
    Applying transformation on $u$,
    \[ T(u) = T\left( \sum_{j=1}^{n} c_j v_j \right) = \sum_{j=1}^{n} c_j T(v_j) \tag{1} \]
    Also by the definition of matrix representation we can write,
    \[ T(v_j) = \sum_{i=1}^{m} A_{ij} w_i \quad \text{ for } 1 \le j \le n. \]
    Then,
    \[ [T]_\beta^\gamma = A \]
    Again,
    \begin{align*}
    (1) \implies T(u) &= \sum_{j=1}^{n} c_j T(v_j) = \sum_{j=1}^{n} c_j \left( \sum_{i=1}^{m} A_{ij} w_i \right) \\
    &= \sum_{i=1}^{m} \left( \sum_{j=1}^{n} A_{ij} c_j \right) w_i
    \end{align*}
    \[ \therefore \ T(u) = \left( \sum_{j=1}^{n} A_{1j} c_j \right) w_1 + \left( \sum_{j=1}^{n} A_{2j} c_j \right) w_2 + \dots + \left( \sum_{j=1}^{n} A_{mj} c_j \right) w_m \]
    Since, $w_i \in \gamma$. Then,
    \[ [T(u)]_\gamma = \begin{bmatrix} \displaystyle\sum_{j=1}^{n} A_{1j} c_j \\ \displaystyle\sum_{j=1}^{n} A_{2j} c_j \\ \vdots \\ \displaystyle\sum_{j=1}^{n} A_{mj} c_j \end{bmatrix} = \begin{bmatrix} A_{11} & A_{12} & \dots & A_{1n} \\ A_{21} & A_{22} & \dots & A_{2n} \\ \vdots & \vdots & \ddots & \vdots \\ A_{m1} & A_{m2} & \dots & A_{mn} \end{bmatrix} \begin{bmatrix} c_1 \\ c_2 \\ \vdots \\ c_n \end{bmatrix} = A \cdot [u]_\beta = [T]_\beta^\gamma [u]_\beta \]
    \[ \therefore \ [T(u)]_\gamma = [T]_\beta^\gamma [u]_\beta \]
\end{proof}

\begin{definition}
    Let $A$ be an $m \times n$ matrix with entries from field $F$. We denote by $L_A$ the mapping $L_A:F^n \to F^m$ defined by $L_A(x) = Ax$ (the matrix product of $A$ and $x$) for each column vector $x \in F^n$. We call $L_A$ a left-multiplication transformation.
\end{definition}

\begin{theorem}
    Let $A$ be an $m \times n$ matrix with entries from $F$. Then $L_A : F^n \to F^m$ is linear. Also if $B$ is any other $m \times n$ matrix (with entries from $F$) and $\beta$ and $\gamma$ are the standard ordered bases for $F^n$ and $F^m$, respectively, then the following is true:

\begin{enumerate}
    \item $[L_A]_\beta^\gamma = A$
    \item $L_A = L_B$ if and only if $A = B$
    \item $L_{A+B} = L_A + L_B$ and $L_{aA} = aL_A$ for all $a \in F$.
    \item If $T: F^n \to F^m$ is linear, then there exists a unique $m \times n$ matrix $C$ such that $T = L_C$. In fact, $C = [T]_\beta^\gamma$.
    \item If $E$ is an $n \times p$ matrix, then $L_{AE} = L_A \cdot L_E$.
    \item If $m = n$ then, $L_{I_n} = I_{F^n}$
\end{enumerate}
\end{theorem}

\begin{proof}
    \begin{enumerate}
        \item The $j$-th column of $[L_A]_\beta^\gamma$ is equal to $L_A(e_j)$. However $L_A(e_j) = Ae_j$, which is also the $j$-th column of $A$, by Theorem 3.3(2). Thus, $[L_A]_\beta^\gamma = A$.

        \item If $L_A = L_B$, from part (1), $A = [L_A]_\beta^\gamma = [L_B]_\beta^\gamma = B$. Also if $A = B$ then, $L_A(x) = Ax = Bx = L_B(x)$ for all $x \in F^n$.

        \item $L_{A+B}(x) = (A+B)x = Ax + Bx = L_A(x) + L_B(x)$ for $x \in F^n$. \\
        Again, $L_{aA}(x) = (aA)x = a(Ax) = aL_A(x)$ for all $a \in F$ and $x \in F^n$.

        \item Let $C = [T]_\beta^\gamma$. By Theorem 3.5, $[T(x)]_\gamma = [T]_\beta^\gamma [x]_\beta$, or $T(x) = Cx = L_C(x)$ for all $x \in F^n$. So $T = L_C$. The uniqueness of $C$ follows from (2).

        \item Since $A$ is an $m \times n$ matrix and $E$ is an $n \times p$ matrix then $AE$ is a matrix of order $m \times p$. For any $j$ ($1 \le j \le p$), we can say that $(AE)e_j$ is the $j$-th column of $AE$ which is also equal to $A(Ee_j)$. Thus, $(AE)e_j = A(Ee_j)$. Thus,
        \[ L_{AE}(e_j) = (AE)e_j = A(Ee_j) = A \cdot L_E(e_j) = L_A(L_E(e_j)). \]
        Hence, $L_{AE} = L_A \cdot L_E$.

        \item For any $x \in F^n$, $L_{I_n}(x) = I_n x = x$ and again, $I_{F^n}(x) = x$. Thus, $L_{I_n}(x) = x = I_{F^n}(x)$. Thus,
        \[ L_{I_n} = I_{F^n} \]
    \end{enumerate}
\end{proof}

\begin{theorem}
    Let $A, B$ and $C$ be matrices such that $A(BC)$ is defined. Then $(AB)C$ is also defined and $A(BC) = (AB)C$; i.e., matrix multiplication is associative.
\end{theorem}

\begin{proof}
    Let $B$ be an $m \times n$ matrix. We have $A(BC)$ is defined. Then $BC$ is defined. Thus $C$ must have the no. of rows equal to the no. of columns of $B$. $C$ must have $n$ rows. Let us say $C$ is an $n \times p$ matrix. Again $A$ and $BC$ must be defined. Here $BC$ must be an $m \times p$ matrix and thus $A$ must have $m$ columns. Let $A$ be an $l \times m$ matrix. 

    Thus so far we have:
    \begin{itemize}
        \item $A$ is an $l \times m$ matrix,
        \item $B$ is an $m \times n$ matrix,
        \item $C$ is an $n \times p$ matrix.
    \end{itemize}

    Thus, $AB$ must be an $l \times n$ matrix. Thus $AB$ is defined and $C$ has the exact no. of rows to that of columns of $AB$. Thus $(AB)C$ is defined and must be a matrix of order $l \times p$.

    Using Theorem 3.6 (5) and Theorem 2.2 (2) we get,
    \[ L_{A(BC)} = L_A \cdot L_{BC} = L_A(L_B L_C) = (L_A L_B)L_C = L_{AB} L_C = L_{(AB)C} \]
    So from Theorem 3.6 (2), we get:
    \[ A(BC) = (AB)C \]
\end{proof}

\section{Conclusion}
Thus throughout the study, we establish matrix multiplication aren't simple computational rules but the relationship between composition of transformations and matrix multiplication forms one of the core foundations of linear algebra. By proving that the coordinate matrix of a composed transformation $UT$ is exactly the product of their individual matrices, $[U]_\beta^\gamma [T]_\alpha^\beta$, we bridge the abstract linear transformation vector spaces with the algebra of matrices.
